### 1.4 –°–∫—Ä–∏–ø—Ç –∞–≤—Ç–æ–º–∞—Ç–∏–∑–∞—Ü–∏–∏ —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏—è

#!/usr/bin/env python3
"""
–°–∫—Ä–∏–ø—Ç –∞–≤—Ç–æ–º–∞—Ç–∏–∑–∞—Ü–∏–∏ —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏—è Cursor CLI –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤
"""
import subprocess

class CursorCLITester:
    def __init__(self, container_name: str = "cursor-agent-life", workspace: str = "/workspace"):
        self.container_name = container_name
        self.workspace = workspace
        self.results_dir = Path("results")
        self.results_dir.mkdir(exist_ok=True)
        
        # –¢–µ—Å—Ç–æ–≤—ã–µ –º–æ–¥–µ–ª–∏
        self.models = [
            "",  # Auto
            "auto",
            "claude-haiku",
            "gpt-4o-mini",
            "gemini-flash",
            "sonnet-3.5",
            "sonnet-4.5",
            "gpt-5",
            "grok"
        ]
        
        # –¢–µ—Å—Ç–æ–≤—ã–µ –ø—Ä–æ–º–ø—Ç—ã
        self.test_prompts = {
            "simple": "–°–æ–∑–¥–∞–π —Ñ–∞–π–ª test.txt —Å —Ç–µ–∫—Å—Ç–æ–º 'Hello'",
            "medium": "–°–æ–∑–¥–∞–π –ø—Ä–æ—Å—Ç—É—é —Ñ—É–Ω–∫—Ü–∏—é –≤–∞–ª–∏–¥–∞—Ü–∏–∏ email –Ω–∞ Python",
            "complex": "–°–æ–∑–¥–∞–π REST API endpoint —Å –∞—É—Ç–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ü–∏–µ–π"
        }
        
        # –ö–æ–º–±–∏–Ω–∞—Ü–∏–∏ —Ñ–ª–∞–≥–æ–≤
        self.flag_combinations = [
            ["-p"],
            ["-p", "--force"],
            ["-p", "--force", "--approve-mcps"],
            ["-p", "--force", "--approve-mcps", "--verbose"]
        ]
    
    def run_command(self, model: str, prompt: str, flags: List[str], timeout: int = 60) -> Dict:
        """–í—ã–ø–æ–ª–Ω–∏—Ç—å –∫–æ–º–∞–Ω–¥—É –∏ –≤–µ—Ä–Ω—É—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç"""
        cmd_parts = ["docker", "exec", self.container_name, "bash", "-c"]
        
        # –§–æ—Ä–º–∏—Ä—É–µ–º –∫–æ–º–∞–Ω–¥—É agent
        agent_cmd = ["cd", self.workspace, "&&", "/root/.local/bin/agent"]
        
        # –î–æ–±–∞–≤–ª—è–µ–º –º–æ–¥–µ–ª—å –µ—Å–ª–∏ —É–∫–∞–∑–∞–Ω–∞
        if model:
            agent_cmd.extend(["-m", model])
        
        # –î–æ–±–∞–≤–ª—è–µ–º —Ñ–ª–∞–≥–∏
        agent_cmd.extend(flags)
        
        # –î–æ–±–∞–≤–ª—è–µ–º prompt
        agent_cmd.append(f'"{prompt}"')
        
        # –û–±—ä–µ–¥–∏–Ω—è–µ–º –≤ bash –∫–æ–º–∞–Ω–¥—É
        bash_cmd = " ".join(agent_cmd)
        cmd_parts.append(bash_cmd)
        
        start_time = time.time()
        result = {
            "model": model or "auto",
            "prompt_type": "simple",  # –û–ø—Ä–µ–¥–µ–ª–∏—Ç—å –∏–∑ prompt
            "flags": " ".join(flags),
            "command": bash_cmd,
            "success": False,
            "return_code": None,
            "duration": None,
            "stdout": "",
            "stderr": "",
            "billing_error": False,
            "timeout": False,
            "error": None
        }
        
        try:
            proc = subprocess.run(
                cmd_parts,
                capture_output=True,
                text=True,
                timeout=timeout,
                encoding='utf-8',
                errors='replace'
            )
            
            result["return_code"] = proc.returncode
            result["stdout"] = proc.stdout[:500]  # –ü–µ—Ä–≤—ã–µ 500 —Å–∏–º–≤–æ–ª–æ–≤
            result["stderr"] = proc.stderr[:500]
            result["success"] = proc.returncode == 0
            result["billing_error"] = "unpaid invoice" in proc.stderr.lower() or "pay your invoice" in proc.stderr.lower()
            result["duration"] = time.time() - start_time
            
        except subprocess.TimeoutExpired:
            result["timeout"] = True
            result["duration"] = timeout
            result["error"] = "Timeout"
        except Exception as e:
            result["error"] = str(e)
        
        return result
    
    def test_all_models(self, prompt_type: str = "simple") -> List[Dict]:
        """–ü—Ä–æ—Ç–µ—Å—Ç–∏—Ä–æ–≤–∞—Ç—å –≤—Å–µ –º–æ–¥–µ–ª–∏ —Å –±–∞–∑–æ–≤–æ–π –∫–æ–º–±–∏–Ω–∞—Ü–∏–µ–π —Ñ–ª–∞–≥–æ–≤"""
        prompt = self.test_prompts[prompt_type]
        flags = ["-p", "--force", "--approve-mcps"]
        
        results = []
        print(f"\n=== –¢–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ –≤—Å–µ—Ö –º–æ–¥–µ–ª–µ–π (prompt: {prompt_type}) ===")
        
        for model in self.models:
            print(f"–¢–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ –º–æ–¥–µ–ª–∏: {model or 'auto'}")
            result = self.run_command(model, prompt, flags)
            results.append(result)
            
            # –í—ã–≤–æ–¥ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞
            status = "‚úÖ" if result["success"] else "‚ùå"
            if result["billing_error"]:
                status = "‚ö†Ô∏è"
            if result["timeout"]:
                status = "‚è≥"
            
            print(f"  {status} –ö–æ–¥: {result['return_code']}, –í—Ä–µ–º—è: {result['duration']:.2f}s")
            
            # –ü–∞—É–∑–∞ –º–µ–∂–¥—É —Ç–µ—Å—Ç–∞–º–∏
            time.sleep(2)
        
        return results
    
    def test_flag_combinations(self, model: str = "claude-haiku") -> List[Dict]:
        """–ü—Ä–æ—Ç–µ—Å—Ç–∏—Ä–æ–≤–∞—Ç—å –≤—Å–µ –∫–æ–º–±–∏–Ω–∞—Ü–∏–∏ —Ñ–ª–∞–≥–æ–≤ —Å –æ–¥–Ω–æ–π –º–æ–¥–µ–ª—å—é"""
        prompt = self.test_prompts["simple"]
        
        results = []
        print(f"\n=== –¢–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ –∫–æ–º–±–∏–Ω–∞—Ü–∏–π —Ñ–ª–∞–≥–æ–≤ (–º–æ–¥–µ–ª—å: {model or 'auto'}) ===")
        
        for flags in self.flag_combinations:
            flags_str = " ".join(flags)
            print(f"–¢–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ —Ñ–ª–∞–≥–æ–≤: {flags_str}")
            
            result = self.run_command(model, prompt, flags)
            result["flags"] = flags_str
            results.append(result)
            
            status = "‚úÖ" if result["success"] else "‚ùå"
            print(f"  {status} –ö–æ–¥: {result['return_code']}, –í—Ä–µ–º—è: {result['duration']:.2f}s")
            
            time.sleep(2)
        
        return results
    
    def save_results(self, results: List[Dict], filename: str):
        """–°–æ—Ö—Ä–∞–Ω–∏—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã –≤ CSV"""
        if not results:
            return
        
        filepath = self.results_dir / filename
        fieldnames = results[0].keys()
        
        with open(filepath, 'w', newline='', encoding='utf-8') as f:
            writer = csv.DictWriter(f, fieldnames=fieldnames)
            writer.writeheader()
            writer.writerows(results)
        
        print(f"\n–†–µ–∑—É–ª—å—Ç–∞—Ç—ã —Å–æ—Ö—Ä–∞–Ω–µ–Ω—ã: {filepath}")
    
    def generate_summary(self, all_results: Dict[str, List[Dict]]) -> Dict:
        """–°–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞—Ç—å —Å–≤–æ–¥–∫—É —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤"""
        summary = {
            "timestamp": datetime.now().isoformat(),
            "total_tests": sum(len(r) for r in all_results.values()),
            "successful": 0,
            "failed": 0,
            "billing_errors": 0,
            "timeouts": 0,
            "working_models": [],
            "working_combinations": [],
            "recommendations": []
        }
        
        # –ê–Ω–∞–ª–∏–∑ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
        for category, results in all_results.items():
            for result in results:
                if result["success"]:
                    summary["successful"] += 1
                    if category == "models" and result["model"] not in summary["working_models"]:
                        summary["working_models"].append(result["model"])
                else:
                    summary["failed"] += 1
                
                if result["billing_error"]:
                    summary["billing_errors"] += 1
                
                if result["timeout"]:
                    summary["timeouts"] += 1
        
        # –†–µ–∫–æ–º–µ–Ω–¥–∞—Ü–∏–∏
        if summary["billing_errors"] > 0:
            summary["recommendations"].append("–ò—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –¥–µ—à–µ–≤—ã–µ –º–æ–¥–µ–ª–∏ (claude-haiku, gpt-4o-mini)")
        
        if summary["timeouts"] > 0:
            summary["recommendations"].append("–£–≤–µ–ª–∏—á–∏—Ç—å —Ç–∞–π–º–∞—É—Ç –∏–ª–∏ –ø—Ä–æ–≤–µ—Ä–∏—Ç—å –∫–æ–Ω—Ç–µ–π–Ω–µ—Ä")
        
        if summary["working_models"]:
            summary["recommendations"].append(f"–†–µ–∫–æ–º–µ–Ω–¥—É–µ–º—ã–µ –º–æ–¥–µ–ª–∏: {', '.join(summary['working_models'][:3])}")
        
        return summary
    
    def run_full_test_suite(self):
        """–ó–∞–ø—É—Å—Ç–∏—Ç—å –ø–æ–ª–Ω—ã–π –Ω–∞–±–æ—Ä —Ç–µ—Å—Ç–æ–≤"""
        print("=" * 80)
        print("–ù–ê–ß–ê–õ–û –ü–û–õ–ù–û–ì–û –¢–ï–°–¢–ò–†–û–í–ê–ù–ò–Ø CURSOR CLI")
